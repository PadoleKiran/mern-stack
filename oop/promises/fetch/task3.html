input.addEventListener("keypress", (e) => { if (e.key === "Enter") { const username = input.value.trim(); if (!username) return; fetch(`https://api.github.com/users/${username}`) .then(res => res.json()) .then(data => { if (data.message === "Not Found") { first.textContent = "❌ User not found!"; return; } first.innerHTML = `
Name: ${data.name || "N/A"}
Username: ${data.login}
Followers: ${data.followers}
Following: ${data.following}
Public Repos: ${data.public_repos}
`; }) .catch(err => { console.error(err); first.textContent = "⚠️ Error fetching data."; }); } });
